Grammar 
@Grammar extends OCL::OCL.grammar
    
      Rules ::= n = Name is = Imports es = Exp* 'end' {
      
        es->iterate(e r = Rules(n).addImports(is) | r.add(e))
        
      }.
      
      Imports ::= 'import' p = RulePath ps = (',' RulePath)* { Seq{p | ps} } | { Seq{} }.
      
      RulePath ::= n = Name ns = ('::' Name)* { Seq{n | ns} }.
      
    end
