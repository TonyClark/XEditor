      Builtin ::= name = BuiltinName '(' args = BuiltinArgs ')' '[' con = BuiltinContext ']' body = Exp 'end' {
        let body = Let(args->collect(arg | ValueBinding(arg,[| values->at(indexOf(arg))>) |])),body) then
            op = Operation("builtin",con,NamedType(),body)
        in [| XRules::Builtin(,) |]
        end
      }.
      
      BuiltinName ::= n = Name { StrExp(n) } | d = Drop { Drop([| StrExp() |]) }.
      
      BuiltinArgs ::= 
        n = AName ns = (',' AName)* { Seq{n|ns} } | { Seq{} }.
      
      BuiltinContext ::=
        rules = Pattern ','
        indent = Pattern ','
        nv = Pattern   ','
        succ = Pattern  ','
        fail = Pattern  ','
        exception = Pattern
        { Seq{rules,Varp("values"),indent,nv,succ,fail,exception} }.
