Package

Overview

Grammar 
@Grammar extends OCL::OCL.grammar
  
    Package ::= 
    
      name = AName 
      
      meta = MetaClass
      
      metaPackage = MetaPackage 
      
      parents = Parents 
      
      imports = PImports
      
      defs = Defs 
      
      'end'
      
      { Package(name,meta,metaPackage,parents,imports,defs) }.
    
    MetaPackage ::= 'metapackage' <> | { null }.
    
    MetaClass ::= 'metaclass' <> | { null }.
    
    Parents ::= 'extends' p = <> ps = (',' <>)* { Seq{p|ps} } | { Seq{} }.
    
    PImports ::= 'import' p = <> ps = (',' <>)* { Seq{p|ps} } | { Seq{} }.
    
    Defs ::= 
    
      <> * | 
      
      '<*' e = <> '*>' { Drop(e) } | 
      
      '<*|' p = <> '|*>' { DropPattern(p) }.
   
  end
